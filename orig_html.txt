        // Função Global de Troca de Tema
        window.toggleTheme = function() {
            const html = document.documentElement;
            const icon = document.querySelector('#theme-toggle i');
            if (html.getAttribute('data-theme') === 'dark') {
                html.setAttribute('data-theme', 'light');
                icon.className = 'fas fa-sun';
            } else {
                html.setAttribute('data-theme', 'dark');
                icon.className = 'fas fa-moon';
            }
        }

        // Função Global de Navegação (Views SPA)
        window.switchSection = function(id, btnElement) {
            // Oculta todas as views
            document.querySelectorAll('.spa-view').forEach(v => v.classList.remove('active'));
            document.getElementById(id).classList.add('active');
            
            // Gerencia ativação no Dock
            document.querySelectorAll('.dock-item').forEach(b => b.classList.remove('active'));
            if(btnElement) btnElement.classList.add('active');
            
            // Se o overlay de liderança estiver aberto e o cara clicar num botão, fecha ele
            if(document.getElementById('cmdOverlay').classList.contains('open')) {
                window.toggleLideranca();
            }
        }

        // Toggle Liderança Overlay
        window.toggleLideranca = function() {
            const overlay = document.getElementById('cmdOverlay');
            if(overlay.classList.contains('open')) {
                overlay.classList.remove('open');
                setTimeout(() => overlay.style.display = 'none', 300); // match transition
            } else {
                overlay.style.display = 'flex';
                // Trigger reflow
                void overlay.offsetWidth;
                overlay.classList.add('open');
            }
        }
